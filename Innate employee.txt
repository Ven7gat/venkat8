 Create the Employee Table:

CREATE TABLE Employee (
 EmpID INT PRIMARY KEY,
 Name VARCHAR(50),
 Department VARCHAR(50),
 Gender VARCHAR(10),
 Salary DECIMAL(10, 2),
 Age INT,
 Experience INT
);

A. Find total salary per department.
Ans: select sum(salary) from employee group by department;

B. List departments where total salary exceeds 200000.
Ans: select department from employee group by department having sum(salary)>200000;

C. Count number of employees in each department.
Ans: select count(name) from employee group by department;

D. List departments with more than 3 employees.
Ans: select department from employee group by department having count(name)>3;

E. Find average salary by gender.
Ans: select avg(salary) from employee group by gender;

F. Show gender-wise employee count, only if count is more than 5.
Ans:  select count(name) from employee group by gender having count(name)>5;

G. List departments with average salary above 60000.
Ans: select department from employee group by department having avg(salary)>60000;

H. List number of male and female employees per department.
Ans: select sum(case when gender='male' then 1 else 0 end) as malecount, sum( case when gender='female' then 1 else 0 end) as femalecount from employee group by department;

I. Find departments where the average experience is more than 7 years.
Ans: select department from employee group by department having avg(experience)>7;

J. List departments where the max salary is above 70000.
Ans: select department from employee group by department having max(salary)>70000;

K. Find average age by department.
Ans: select avg(age),department from employee group by department;

L. List all departments where female employees earn more than 60000 on average.
Ans: SELECT department FROM employee WHERE gender = 'female' GROUP BY department HAVING AVG(salary) > 60000;

M. Find departments with total experience greater than 20 years.
Ans: select department from employee group by department having sum(experience)>20;

N. Find gender-wise average experience per department.
Ans: select gender,avg(experience) from employee group by gender,experience;

O. List departments where average age is under 30.
Ans: select department from employee group by department having avg(age)<30;

P. Find departments where more than one female is working.
Ans: select department,count(gender) from employee where gender='female' group by department having count(gender)>1 ;

Q. Find departments where both male and female employees exist.
Ans: select department from employee group by department having count(distinct gender)=2;

R. List departments with highest average experience.
Ans: select department,avg(experience) as avg from employee group by department order by avg desc limit 3; 

S. Find gender and department combinations where total salary is above 100000.
Ans:  select gender,department from employee group by gender,department having sum(salary)>100000;

T. Show department-wise count of employees under age 35.
Ans: select department,count(EmpID) from employee where age<35 group by department;

U. List top 3 departments with highest total salary.
Ans: select department,sum(salary) from employee group by department order by sum(salary) desc limit 3;

V. List departments where all employees have more than 5 years of experience.
Ans: select department from employee group by department having min(experience)>5;

W. Find departments where at least one employee earns less than 55000.
Ans: SELECT department FROM employee GROUP BY department HAVING MIN(salary) < 55000;
